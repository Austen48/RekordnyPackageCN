#更新党派支持度修正
Update_BCF_party_support = {
	if = {
		limit = {
			NOT = {
				check_variable = {
					marie_support_change = 0
				}
			}
		}
		custom_effect_tooltip = Marie_support_scripted_effect_change
		add_to_variable = {
			bcf_marie_support = marie_support_change
		}
	}
	if = {
		limit = {
			NOT = {
				check_variable = {
					oshida_support_change = 0
				}
			}
		}
		custom_effect_tooltip = Oshida_support_scripted_effect_change
		add_to_variable = {
			bcf_oshida_support = oshida_support_change
		}
	}
	if = {
		limit = {
			NOT = {
				check_variable = {
					andou_support_change = 0
				}
			}
		}
		custom_effect_tooltip = Andou_support_scripted_effect_change
		add_to_variable = {
			bcf_andou_support = andou_support_change
		}
	}

}
#添加所有BCF内战国家的BCF核心
BCF_add_civil_warloard_core = {
	every_country = {
		limit = {
			OR = {
				tag = B01
				tag = B02
				tag = BRE
				tag = AQU
				tag = MGC
				tag = ORL
			}
		}
		every_state = {
			limit = {
				is_core_of = BCF
			}
			add_core_of = PREV
		}
	}
}
#移除所有BCF内战国家的BCF核心
BCF_remove_civil_warloard_core = {
	every_country = {
		limit = {
			OR = {
				tag = B01
				tag = B02
				tag = BRE
				tag = AQU
				tag = MGC
				tag = ORL
			}
		}
		every_state = {
			limit = {
				is_core_of = BCF
			}
			remove_core_of = PREV
		}
	}
}
